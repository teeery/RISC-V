# 公共类型定义

> 本文档定义所有模块共享的基础类型。各模块的头文件 `#include "types.h"` 即可使用。

---

## 1. 基础类型

```c
#ifndef TYPES_H
#define TYPES_H

#include <stdint.h>
#include <stdbool.h>
#include <stddef.h>

// ============================================================
// 权限级别
// ============================================================
typedef enum {
    PRIV_USER       = 0,  // U-mode: 用户态
    PRIV_SUPERVISOR = 1,  // S-mode: 监管态（保留，本阶段基本不用）
    PRIV_MACHINE    = 3,  // M-mode: 机器态（模拟器始终运行在此级别）
} PrivilegeLevel;

// ============================================================
// 异常类型（RISC-V Privileged Spec 定义）
// ============================================================
typedef enum {
    EXC_NONE                   = 0,
    EXC_INST_ADDR_MISALIGNED   = 1,   // 指令地址未对齐
    EXC_INST_ACCESS_FAULT      = 2,   // 取指访问错误
    EXC_ILLEGAL_INST           = 3,   // 非法指令
    EXC_BREAKPOINT             = 4,   // 断点 (ebreak)
    EXC_LOAD_ADDR_MISALIGNED   = 5,   // Load 地址未对齐
    EXC_LOAD_ACCESS_FAULT      = 6,   // Load 访问错误
    EXC_STORE_ADDR_MISALIGNED  = 7,   // Store 地址未对齐
    EXC_STORE_ACCESS_FAULT     = 8,   // Store 访问错误
    EXC_ECALL_U                = 9,   // ECALL from U-mode
    EXC_ECALL_S                = 10,  // ECALL from S-mode (保留)
    EXC_ECALL_M                = 11,  // ECALL from M-mode
    EXC_PAGE_FAULT_INST        = 12,  // 取指页错误
    EXC_PAGE_FAULT_LOAD        = 13,  // Load 页错误
    EXC_PAGE_FAULT_STORE       = 15,  // Store 页错误
} ExceptionType;

// ============================================================
// 内存权限标志（PhysicalMemory 层用）
// ============================================================
#define MEM_READ   (1 << 0)
#define MEM_WRITE  (1 << 1)
#define MEM_EXEC   (1 << 2)

// ============================================================
// PTE 权限标志（MMU 页表层用）
// ============================================================
#define PTE_VALID    (1 << 0)
#define PTE_READ     (1 << 1)
#define PTE_WRITE    (1 << 2)
#define PTE_EXEC     (1 << 3)
#define PTE_USER     (1 << 4)
#define PTE_GLOBAL   (1 << 5)
#define PTE_ACCESSED (1 << 6)
#define PTE_DIRTY    (1 << 7)

// ============================================================
// MEM 权限标志 → PTE 权限标志 转换（Loader 等同时操作两层时使用）
// ============================================================
// 背景：MEM_* 和 PTE_* 的 bit 位置不同（MEM_R=1<<0, PTE_R=1<<1），
// Loader 拿到 ELF 段的 p_flags 后转成 MEM_* 调 mem_map，
// 再通过此函数转为 PTE_* 调 mmu_map_page。
static inline uint8_t mem_perm_to_pte_flags(uint8_t mem_flags) {
    uint8_t pte = PTE_VALID;  // 所有映射默认有效
    if (mem_flags & MEM_READ)  pte |= PTE_READ;
    if (mem_flags & MEM_WRITE) pte |= PTE_WRITE;
    if (mem_flags & MEM_EXEC)  pte |= PTE_EXEC;
    return pte;
}

// ============================================================
// Sv32 页表常量
// ============================================================
#define PAGE_SIZE    4096
#define PAGE_SHIFT   12

// ============================================================
// satp CSR 模式
// ============================================================
#define SATP_MODE_OFF  0   // Bare 模式：地址不翻译（vaddr == paddr）
#define SATP_MODE_SV32 1   // Sv32 模式：两级页表翻译

#endif // TYPES_H
```

## 2. 各模块依赖关系

```
                    ┌─────────────┐
                    │   types.h   │  ← 零依赖，所有模块的基础
                    └──────┬──────┘
           ┌───────────────┼───────────────┐
           ▼               ▼               ▼
    ┌──────────┐    ┌──────────┐    ┌──────────┐
    │ memory.h │    │  cpu.h   │    │ decode.h │
    │(Physical │    │(CPU 结构体)│   │(位域宏)  │
    │ Memory)  │    └────┬─────┘    └────┬─────┘
    └────┬─────┘         │              │
         │               └──────┬───────┘
         ▼                      ▼
    ┌──────────┐          ┌──────────┐
    │  mmu.h   │          │execute.h │
    │(MMUState)│          └────┬─────┘
    └────┬─────┘               │
         │                     │
         └─────────┬───────────┘
                   ▼
            ┌──────────┐
            │simulator.h│ ← 顶层聚合所有模块
            └──────────┘
	    ┌────────────┼────────────┐
	    ▼            ▼            ▼
	┌────────┐ ┌──────────┐ ┌──────────┐
	│loader.h│ │debugger.h│ │   main.c │ ← 使用 Simulator 运行
	└────────┘ └──────────┘ └──────────┘
```

**依赖规则：**
- 上层可以 `#include` 下层，下层**绝不** include 上层
- `types.h` 是被所有人 include 的叶子节点
- `memory.h` 不依赖 `mmu.h`（PhysicalMemory 不知道自己被 MMU 用）
- `mmu.h` 依赖 `memory.h`（MMU 翻译后调用 PhysicalMemory）
- `cpu.h` 只依赖 `types.h`（CPU 结构体不包含 MMU 指针）
- `execute.h` 依赖 `types.h`（需要 `ExceptionType`）
- CPU 通过 `Simulator*` 访问 `mmu` 和 `pmem`，不直接 include 它们
- syscall 处理内联在 `cpu_execute()` 中，不独立为模块（团队结论 5）
- `simulator.h` include 所有模块头文件，对外提供 `Simulator` 结构体

> **注意**：此图描述的是 **C 编译期的 `#include` 依赖关系**（头文件之间的编译依赖）。各模块运行时通过 `Simulator*` 指针互相访问（如 Loader 调用 `mmu_map_page`、Debugger 读取 `cpu.regs[]`），运行时调用关系是另一层概念——真正的运行时调用关系以各模块自己的设计文档为准。

## 3. Simulator 顶层结构体

```c
// simulator.h

#ifndef SIMULATOR_H
#define SIMULATOR_H

#include "types.h"
#include "memory.h"
#include "mmu.h"
#include "cpu.h"

// ============================================================
// 断点定义
// ============================================================
typedef struct {
    uint32_t addr;              // 断点地址
    uint32_t original_insn;     // 被替换前的原始指令
    bool     enabled;           // 是否启用
} Breakpoint;

// ============================================================
// 模拟器顶层结构体 — 聚合所有模块状态
// ============================================================
typedef struct {
    // 核心模块
    CPU             cpu;           // CPU 状态（寄存器、PC、CSR）  → 李特
    MMUState        mmu;           // MMU 状态（页表、satp）       → 焕聪
    PhysicalMemory  pmem;          // 物理内存                      → 焕聪

    // 调试相关
    Breakpoint     *breakpoints;   // 断点数组（动态分配）
    int             bp_count;      // 当前断点数量
    int             bp_capacity;   // 断点数组容量
    bool            single_step;   // 单步执行标志
    bool            debug_mode;    // 调试模式（Debugger 接管时设为 true）

    // 统计
    uint64_t        inst_count;    // 已执行指令数
    uint64_t        cycle_count;   // 周期计数（基础版 = inst_count）
} Simulator;

// 初始化 / 清理
void sim_init(Simulator *sim);
void sim_destroy(Simulator *sim);

// 主执行循环
void sim_run(Simulator *sim);

// 执行单条指令
void sim_step(Simulator *sim);

// 加载 ELF
bool sim_load_elf(Simulator *sim, const char *path);

#endif // SIMULATOR_H
```

**关于断点位置的说明：**
- 断点数组放在 `Simulator` 而非 `Debugger` 内部。原因：
  1. CPU 执行 `ebreak` 时需要查断点列表区分"断点触发"还是"程序自身的 ebreak"
  2. 如果放在 Debugger 内部，CPU 需要反向依赖 Debugger，破坏依赖方向
  3. Debugger 模块通过 `Simulator*` 指针操作断点数组，符合单向依赖

## 4. 最终接口签名规范（已对齐）

### 4.1 返回值约定

| 场景 | 返回类型 | 含义 |
|------|---------|------|
| 正常操作（读/写/映射） | `bool` | `true` = 成功，`false` = 失败 |
| 查找类（找空闲区域/查 brk） | `uint32_t` | 返回地址，`0` 表示失败（注意：`brk(0)` 是查询操作，返回当前 brk 值，不为 0） |
| 错误详情 | `ExceptionType*` 输出参数 | 调用者传入 `&exc`，失败时被填充 |

### 4.2 函数名前缀约定

| 前缀 | 所属层 | 操作对象 |
|------|--------|---------|
| `mem_` | PhysicalMemory 层 | 物理地址 `paddr` |
| `mmu_` | MMU 虚拟内存层 | 虚拟地址 `vaddr` |

### 4.3 参数顺序约定

```
<前缀>_<操作>_<位宽>(<自身状态>, <被操作对象>, <地址>, <数据>, <权限/异常>)
```

- 第一个参数：**自身状态指针**（`MMUState*` 或 `PhysicalMemory*`）
- 如果是 MMU 函数，第二个参数：**PhysicalMemory 指针**（MMU 翻译后操作的目标）
- 地址参数在数据参数之前
- `PrivilegeLevel` 和 `ExceptionType*` 放在最后

### 4.4 CPU 调用的完整 Memory 接口

以下是 **李特（CPU）在 `cpu_execute` 中实际调用的函数签名**：

```c
// ====== MMU 层（CPU 只调这一层，不直接调 mem_*）======

// 初始化
void mmu_init(MMUState *mmu);

// 地址翻译（CPU 一般不直接调，由 mmu_read/write 内部调用）
bool mmu_translate(MMUState *mmu, uint32_t vaddr, uint32_t *paddr,
                   bool is_write, bool is_exec, PrivilegeLevel priv,
                   ExceptionType *exc);

// 虚拟地址读写 —— CPU 取指、LOAD/STORE 用
bool mmu_read_8 (MMUState *mmu, PhysicalMemory *pmem, uint32_t vaddr,
                 uint8_t  *val, PrivilegeLevel priv, ExceptionType *exc);
bool mmu_read_16(MMUState *mmu, PhysicalMemory *pmem, uint32_t vaddr,
                 uint16_t *val, PrivilegeLevel priv, ExceptionType *exc);
bool mmu_read_32(MMUState *mmu, PhysicalMemory *pmem, uint32_t vaddr,
                 uint32_t *val, PrivilegeLevel priv, ExceptionType *exc);

bool mmu_write_8 (MMUState *mmu, PhysicalMemory *pmem, uint32_t vaddr,
                  uint8_t  val, PrivilegeLevel priv, ExceptionType *exc);
bool mmu_write_16(MMUState *mmu, PhysicalMemory *pmem, uint32_t vaddr,
                  uint16_t val, PrivilegeLevel priv, ExceptionType *exc);
bool mmu_write_32(MMUState *mmu, PhysicalMemory *pmem, uint32_t vaddr,
                  uint32_t val, PrivilegeLevel priv, ExceptionType *exc);

// 批量读写 —— syscall 用
bool mmu_read (MMUState *mmu, PhysicalMemory *pmem, uint32_t vaddr,
               uint8_t *buf, uint32_t len, PrivilegeLevel priv, ExceptionType *exc);
bool mmu_write(MMUState *mmu, PhysicalMemory *pmem, uint32_t vaddr,
               const uint8_t *buf, uint32_t len, PrivilegeLevel priv, ExceptionType *exc);

// 页映射 —— Loader 用
bool mmu_map_page(MMUState *mmu, uint32_t vaddr, uint32_t paddr, uint8_t flags);

// 堆管理 —— syscall brk 用（由 MMU 层转发到 PhysicalMemory 层）
// 设计意��：CPU 只调 mmu_*，不直接调 mem_*。brk 虽与地址翻译无关，
// 但为保持规则绝对性，通过此包装函数间接调用 mem_brk。
uint32_t mmu_brk(MMUState *mmu, PhysicalMemory *pmem, uint32_t new_brk);
// brk(0) 表示查询当前值，返回当前 brk 地址（不为 0）。

// ====== PhysicalMemory 层（CPU 不直接调，由 MMU / Loader / Debugger 内部调用）======

void     mem_init(PhysicalMemory *pmem, uint32_t size);
void     mem_destroy(PhysicalMemory *pmem);

bool     mem_read_8 (PhysicalMemory *pmem, uint32_t addr, uint8_t  *val);
bool     mem_read_16(PhysicalMemory *pmem, uint32_t addr, uint16_t *val);
bool     mem_read_32(PhysicalMemory *pmem, uint32_t addr, uint32_t *val);

bool     mem_write_8 (PhysicalMemory *pmem, uint32_t addr, uint8_t  val);
bool     mem_write_16(PhysicalMemory *pmem, uint32_t addr, uint16_t val);
bool     mem_write_32(PhysicalMemory *pmem, uint32_t addr, uint32_t val);

bool     mem_map(PhysicalMemory *pmem, uint32_t base, uint32_t size,
                 uint8_t flags, const char *name);
uint32_t mem_find_free(PhysicalMemory *pmem, uint32_t size, uint32_t align);
uint32_t mem_brk(PhysicalMemory *pmem, uint32_t new_brk);
bool     mem_load(PhysicalMemory *pmem, uint32_t addr, const uint8_t *data,
                  uint32_t size);
void     mem_dump(PhysicalMemory *pmem, uint32_t addr, uint32_t len);
```
